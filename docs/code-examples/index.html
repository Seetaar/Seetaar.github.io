    Примеры использования плагина Code Include

    Плагин mkdocs-codeinclude-plugin позволяет включать код из внешних файлов прямо в документацию.

    Демонстрация включения кода

    Эта страница демонстрирует различные способы включения кода из внешних файлов с помощью пользовательских макросов.

    Пример 1: Включение всего файла

    Включение всего Python-файла целиком:

    #!/usr/bin/env python3
&quot;&quot;&quot;
Calculator Module - A simple calculator with basic operations.

This module provides basic arithmetic operations and maintains a history
of calculations. It demonstrates different ways to include code snippets
in documentation.
&quot;&quot;&quot;

from datetime import datetime
from typing import List, Dict, Optional


class Calculator:
    &quot;&quot;&quot;
    A simple calculator class that performs basic arithmetic operations.

    Attributes:
        history (List[Dict]): List of performed operations with results.
        name (str): Name of the calculator instance.
    &quot;&quot;&quot;

    def __init__(self, name: str = &quot;Default Calculator&quot;):
        &quot;&quot;&quot;
        Initialize a new Calculator instance.

        Args:
            name (str): Optional name for this calculator instance.
        &quot;&quot;&quot;
        self.history: List[Dict] = []
        self.name = name

    # &lt;-- add-method
    def add(self, a: float, b: float) -&gt; float:
        &quot;&quot;&quot;
        Add two numbers together.

        Args:
            a (float): First number.
            b (float): Second number.

        Returns:
            float: Sum of a and b.
        &quot;&quot;&quot;
        result = a + b
        self._save_to_history(&quot;add&quot;, a, b, result)
        return result

    # --&gt; add-method

    # &lt;-- subtract-method
    def subtract(self, a: float, b: float) -&gt; float:
        &quot;&quot;&quot;
        Subtract second number from first number.

        Args:
            a (float): First number.
            b (float): Second number to subtract.

        Returns:
            float: Difference between a and b.
        &quot;&quot;&quot;
        result = a - b
        self._save_to_history(&quot;subtract&quot;, a, b, result)
        return result

    # --&gt; subtract-method

    # &lt;-- multiply-method
    def multiply(self, a: float, b: float) -&gt; float:
        &quot;&quot;&quot;
        Multiply two numbers.

        Args:
            a (float): First number.
            b (float): Second number.

        Returns:
            float: Product of a and b.
        &quot;&quot;&quot;
        result = a * b
        self._save_to_history(&quot;multiply&quot;, a, b, result)
        return result

    # --&gt; multiply-method

    # &lt;-- divide-method
    def divide(self, a: float, b: float) -&gt; Optional[float]:
        &quot;&quot;&quot;
        Divide first number by second number.

        Args:
            a (float): Numerator.
            b (float): Denominator.

        Returns:
            Optional[float]: Quotient of a divided by b.
                            Returns None if b is zero.
        &quot;&quot;&quot;
        if b == 0:
            print(&quot;Error: Division by zero&quot;)
            return None
        result = a / b
        self._save_to_history(&quot;divide&quot;, a, b, result)
        return result

    # --&gt; divide-method

    # &lt;-- save-method
    def _save_to_history(self, operation: str, a: float, b: float, result: float) -&gt; None:
        &quot;&quot;&quot;
        Save calculation to history.

        Args:
            operation (str): Name of the operation performed.
            a (float): First operand.
            b (float): Second operand.
            result (float): Result of the operation.
        &quot;&quot;&quot;
        self.history.append({
            &#x27;operation&#x27;: operation,
            &#x27;operands&#x27;: (a, b),
            &#x27;result&#x27;: result,
            &#x27;timestamp&#x27;: datetime.now().isoformat(),
            &#x27;calculator&#x27;: self.name
        })

    # --&gt; save-method

    def get_history(self) -&gt; List[Dict]:
        &quot;&quot;&quot;
        Return the complete calculation history.

        Returns:
            List[Dict]: List of all performed operations with their details.
        &quot;&quot;&quot;
        return self.history

    def clear_history(self) -&gt; None:
        &quot;&quot;&quot;Clear all calculation history.&quot;&quot;&quot;
        self.history.clear()


# &lt;-- helper-function
def format_result(result: float, precision: int = 2) -&gt; str:
    &quot;&quot;&quot;
    Format a numeric result with specified precision.

    Args:
        result (float): Number to format.
        precision (int): Number of decimal places. Defaults to 2.

    Returns:
        str: Formatted number as string.
    &quot;&quot;&quot;
    return f&quot;{result:.{precision}f}&quot;


# --&gt; helper-function


# &lt;-- usage-example
if __name__ == &quot;__main__&quot;:
    # Create a calculator instance
    calc = Calculator(&quot;My Calculator&quot;)

    # Perform some calculations
    sum_result = calc.add(10, 5)
    print(f&quot;Sum: {sum_result}&quot;)

    difference = calc.subtract(10, 5)
    print(f&quot;Difference: {difference}&quot;)

    product = calc.multiply(10, 5)
    print(f&quot;Product: {product}&quot;)

    quotient = calc.divide(10, 5)
    print(f&quot;Quotient: {quotient}&quot;)

    # Show history
    print(&quot;\nCalculation History:&quot;)
    for entry in calc.get_history():
        print(f&quot;  {entry[&#x27;operation&#x27;]}: {entry[&#x27;operands&#x27;]} = {entry[&#x27;result&#x27;]}&quot;)
# --&gt; usage-example
